# I/O and Files

## Reading a file

```mg
use std::fs;

pub fn main() / io {
    // Read entire file to string
    let content = fs::read("README.md")?;
    println!("File has {} bytes", content.len());

    // Read as bytes
    let bytes = fs::read_bytes("image.png")?;
}
```

## Writing a file

```mg
use std::fs;

pub fn main() / io {
    fs::write("output.txt", "Hello, world!")?;

    // Append
    fs::append("log.txt", "new entry\n")?;
}
```

## Buffered I/O

For large files, use buffered readers and writers:

```mg
use std::io::{File, BufReader, BufWriter, Read, Write};

pub fn process_large_file(path: &str) / io {
    let file = File::open(path)?;
    let reader = BufReader::new(file);

    let mut line = String::new();
    loop {
        let bytes_read = reader.read_line(&mut line)?;
        if bytes_read == 0 { break; }
        process(&line);
        line.clear();
    }
}
```

## The Read and Write traits

Implement `Read` or `Write` for custom types:

```mg
use std::io::{Read, Write};

struct Counter {
    count: usize,
}

impl Read for Counter {
    pub fn read(&mut self, buf: &mut [u8]) -> Result<usize, IoError> / io {
        let data = format!("{}", self.count).as_bytes();
        let len = data.len().min(buf.len());
        buf[..len].copy_from_slice(&data[..len]);
        self.count += 1;
        Ok(len)
    }
}
```

## Directory operations

```mg
use std::fs;

pub fn main() / io {
    // Create directories
    fs::create_dir("output")?;
    fs::create_dir_all("deep/nested/path")?;

    // List directory contents
    let entries = fs::read_dir("src")?;
    for entry in entries {
        println!("{} - {}", entry.name(), entry.file_type());
    }

    // Recursive traversal
    let walker = fs::walk("project");
    for entry in walker {
        if entry.is_file() {
            println!("File: {}", entry.path());
        }
    }

    // File metadata
    let meta = fs::metadata("main.mg")?;
    println!("Size: {} bytes", meta.size());
    println!("Modified: {}", meta.modified());
}
```

## Standard streams

```mg
use std::io::{stdin, stdout, stderr};

pub fn main() / io {
    // Read from stdin
    let input = stdin().read_line()?;

    // Write to stdout explicitly
    stdout().write("prompt> ")?;
    stdout().flush()?;

    // Write to stderr
    stderr().write("warning: something happened\n")?;
}
```
